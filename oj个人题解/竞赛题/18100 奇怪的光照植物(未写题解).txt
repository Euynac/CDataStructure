#include<stdio.h>
#include<string.h>
void shaisun(int p[],int L,int R,int a)
{
    int i;
    for(i=L;i<=R;i++)
        p[i]+=a;
}
int main()
{
    int T,n,m,i,j,sum[10]={};//T次实验，每次n棵植物，每次进行m次操作，每次实验的结果；
    int L,R,a;
    int plant[100000]={};
    scanf("%d",&T);
    for(i=0;i<T;i++)
    {
        scanf("%d%d",&n,&m);
        memset(plant,0,sizeof(plant));
        for(j=0;j<m;j++)
        {
            scanf("%d%d%d",&L,&R,&a);
            shaisun(plant,L,R,a);
        }
        for(j=0;j<n;j++)
        {
            if(plant[j]%2==1)
                sum[i]+=plant[j];
        }
    }
    for(i=0;i<T;i++)
        printf("%d\n",sum[i]);
    return 0;
}

#include<stdio.h>
#include<string.h>

int main()
{
    int d[100005]={};
    int l,r,a,i;
    int T,n,m;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%d",&n,&m);
        memset(d,0,sizeof(d));
        for(i=1;i<=m;i++)
        {
            scanf("%d%d%d",&l,&r,&a);
            d[l]+=a;
            d[r+1]-=a;
        }
        int ans=0,count=0;
        for(i=0;i<n;i++)
        {
            count+=d[i];
            if(count%2==1)
                ans+=count;
        }
        printf("%d\n",ans);
    }
    return 0;
}
