#include <stdlib.h>
struct node
{
    int num;
    int label;
    struct node *next;
};
typedef struct node D;

D *create(int n)
{
    D *head,*tail,*p;
    int i,j;
    head=NULL;
    for(i=0,j=1;i<n;i++,j++)
    {
        if(j==4)
            j=1;
        p=(D*)malloc(sizeof(D));
        p->num=i+1;
        p->label=j;
        p->next=NULL;
        if(head==NULL)
            head=p;
        else
            tail->next=p;
        tail=p;
    }
    return(head);
}
