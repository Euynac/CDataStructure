#include<stdio.h>
int main()
{
    int a[20],i,j,count=0;
    for(i=0;i<20;i++)
    {
        scanf("%d",&a[i]);
    }
    for(j=0;j<20;j++)
    {
        for(i=j+1;i<20;i++)
        {
            if(a[i]==0)
                continue;
            if(a[j]==a[i])
            {
                count++;
                a[i]=0;
            }
        }
    }
    if(count==19)
       count=20;
    printf("%d",20-count);
    return 0;
}
